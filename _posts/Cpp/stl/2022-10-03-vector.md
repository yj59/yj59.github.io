#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include<queue>
using namespace std;
int n, m, i;
queue<int> q;
int main()
{
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; i++) q.push(i);
	printf("<");
	while (q.size() != 0)
	{
		for (i = 1; i < m; i++)
		{
			q.push(q.front());
			q.pop();
		}
		printf("%d", q.front());
		if (q.size() != 1) printf(", ");
		q.pop();
	}
	printf(">");
	return 0;
}
